const formData = new FormData();

formData.append('file', fileContent);


this.http.post(this.SERVICE_URL, formData, {responseType: 'blob'}).subscribe(

(res: Blob) => saveAs(res, 'result.xml'),

(err: String) => console.log('HATA: ', err)
